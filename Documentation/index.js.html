/**
 * @namespace
 * @description 
 * The Validic Mobile library provides a simple way to read and upload data from HealthKit to Validic. The ValidicMobile 
 * plugin can subscribe to specific HealthKit sample types and automatically upload them to Validic in the background as 
 * new data is recorded, even if the app is no longer active.Subscribing to HealthKit updates only needs to be done once 
 * for a user. The subscriptions will be persisted across app launches in the Session object. A typical use of the 
 * HealthKit component would be to create a checkbox or equivalent that adds the required subscriptions when turned on 
 * and removes them when turned off.Healthkit subscriptions are based on sample types. Constants representing each available
 * sample type are defined within
 * */
var ValidicHealthKit = {

    /** */
    /**
     * Set the subscriptions for Validic to observe. Any existing subscriptions will be removed
     * @param {Array<SampleTypes>} subscriptionsArray 
     */
    setSubscriptions(subscriptionsArray) {
        RNValidicHealthkit.setSubscriptions(subscriptionsArray);
    },

    /**
     * @callback FetchHistoryCallback
     * @memberof ValidicHealthKit
     * @param {null|Object} errors - Errors were returned while fetching history
     * @param {Object} results - result summary of historical records fetched
     */

    /**
     * Fetch 180 days of history for the supplied Set types
     * @param {HistoricalSet[]} historicalSets 
     * @param {FetchHistoryCallback} historyCallback 
     */
    fetchHistory(historicalSets, historyCallback) {
        RNValidicHealthkit.fetchHistory(historicalSets, historyCallback);
    },
    /**
     * 
     * @param {HistoricalSet[]} historicalSets 
     * @returns {Promise} - A promise that resolves the results for successful fetch 
     */
    fetchHistoryPromise(historicalSets) {
        return new Promise((resolve, reject) => {
            this.fetchHistory(historicalSets, (errors, results) => {
                if (errors != null) {
                    reject(errors);
                } else {
                    resolve(results);
                }
            })
        });
    },

    /**
     * Subscriptions callback for returning the list of currently subscribed Healthkit data types
     * @callback SubscriptionsCallback
     * @memberof ValidicHealthKit
     * @param {SampleTypes[]} subscriptions - array of SampleTypes currently subscribed to
     * 
     */

    /**
     * 
     * @param {SubscriptionsCallback} subscriptionsCB 
     */
    getCurrentSubscriptions(subscriptionsCB) {
        RNValidicHealthkit.currentSubscriptions(subscriptionsCB);
    },

    /**
     * Get list of currently subscriptions 
     * @returns {Promise<String[]>} returns a promise that resolves an array of currently subscribed data types as strings
     */
    getCurrentSubscriptionsPromise() {
        return new Promise((resolve, _reject) => {
            this.getCurrentSubscriptions((subscriptions) => {
                resolve(subscriptions);
            })
        })
    },
    /**
     * observe Current Subscriptions from the background. This should be called as early as possible to pick up records
     * that were received while the app was launched from the background
     */
    observeCurrentSubscriptions() {
        RNValidicHealthkit.observeCurrentSubscriptions();
    },
